% CSL References (required for Quarto bibliography)
% ------------------------------------------------
% Quarto generates \bibitem[\citeproctext]{...} where \citeproctext expands to
% nothing, yielding \bibitem[]{...} → empty label "[]".
% Keep \citeproctext defined (empty) as Quarto expects it.
\newcommand{\citeproctext}{}
% Redefine \bibitem to ignore the (empty) optional label and let LaTeX
% auto-number references as [1], [2], …
\let\standardbibitem\bibitem
\renewcommand{\bibitem}[2][]{%
  \standardbibitem{#2}%
}
% CSLReferences is called with two arguments by Quarto: \begin{CSLReferences}{1}{1}
\newenvironment{CSLReferences}[2]{%
  \begin{thebibliography}{#1}%
}{%
  \end{thebibliography}%
}
